**src-tauri/src/widget.rs** — Floating window management:
```rust
use tauri::{AppHandle, WebviewUrl, WebviewWindowBuilder, LogicalPosition, Manager};

pub fn show_widget(app: &AppHandle) -> tauri::Result<()> {
    if app.get_webview_window("recording-widget").is_some() {
        return Ok(()); // idempotent
    }

    let widget = WebviewWindowBuilder::new(
        app,
        "recording-widget",
        WebviewUrl::App("index.html#/widget".into()), // HashRouter route
    )
    .title("")
    .decorations(false)
    .always_on_top(true)
    .resizable(false)
    .inner_size(280.0, 72.0)
    .skip_taskbar(true)
    .build()?;

    // Default position: top-center of primary monitor
    if let Ok(Some(monitor)) = widget.primary_monitor() {
        let screen_size = monitor.size();
        let scale = monitor.scale_factor();
        let x = (screen_size.width as f64 / scale / 2.0) - 140.0;
        let y = 24.0;
        widget.set_position(LogicalPosition::new(x, y))?;
    }

    Ok(())
}

pub fn hide_widget(app: &AppHandle) {
    if let Some(widget) = app.get_webview_window("recording-widget") {
        let _ = widget.close();
    }
}
```

**Position persistence**: The `tauri-plugin-window-state` plugin (registered in Plan 01) automatically persists and restores window positions by label. On second launch, the widget will restore its last position instead of defaulting to top-center. This satisfies the "remembers last position" requirement with zero custom code.

**src/hooks/useRecording.ts** — Central recording state hook:
```typescript
export function useRecording() {
  // State
  const [isRecording, setIsRecording] = useState(false);
  const [isPaused, setIsPaused] = useState(false);
  const [audioLevel, setAudioLevel] = useState(0);
  const [startTime, setStartTime] = useState<number | null>(null);
  const [permissionStatus, setPermissionStatus] = useState<{mic: string, screenRecording: string} | null>(null);

  // Listen for audio-level events from Rust backend
  useEffect(() => {
    const unlisten = listen<number>('audio-level', (e) => setAudioLevel(e.payload));
    return () => { unlisten.then(f => f()); };
  }, []);

  // Listen for recording-toggle events (from tray menu and global shortcut)
  useEffect(() => {
    const unlisten = listen('recording-toggle', () => {
      if (isRecording) {
        stopRecording();
      } else {
        startRecording();
      }
    });
    return () => { unlisten.then(f => f()); };
  }, [isRecording]);

  // Check permissions on mount (for status indicators)
  useEffect(() => {
    invoke<{mic: string, screen_recording: string}>('check_audio_permissions')
      .then(status => setPermissionStatus({mic: status.mic, screenRecording: status.screen_recording}))
      .catch(() => {}); // non-critical
  }, []);

  // 2-hour soft warning
  useEffect(() => {
    if (!isRecording || !startTime) return;
    const twoHours = 2 * 60 * 60 * 1000;
    const elapsed = Date.now() - startTime;
    const remaining = twoHours - elapsed;
    if (remaining <= 0) return;
    const timer = setTimeout(async () => {
      // Use tauri-plugin-notification for native macOS notification
      const { sendNotification } = await import('@tauri-apps/plugin-notification');
      sendNotification({
        title: 'openNotes',
        body: 'Recording has been running for 2 hours. Still going?',
      });
    }, remaining);
    return () => clearTimeout(timer);
  }, [isRecording, startTime]);

  async function startRecording() {
    // JIT permission check (locked decision: request on first recording, not onboarding)
    const hasPermissions = await ensurePermissions();
    if (!hasPermissions) return;

    await invoke('start_recording');
    setIsRecording(true);
    setIsPaused(false);
    setStartTime(Date.now());
  }

  async function stopRecording() {
    const filePath = await invoke<string>('stop_recording');
    setIsRecording(false);
    setIsPaused(false);
    setStartTime(null);
    setAudioLevel(0);
    return filePath;
  }

  async function pauseRecording() {
    await invoke('pause_recording');
    setIsPaused(true);
  }

  async function resumeRecording() {
    await invoke('resume_recording');
    setIsPaused(false);
  }

  return { isRecording, isPaused, audioLevel, startTime, permissionStatus,
           startRecording, stopRecording, pauseRecording, resumeRecording };
}
```

**Permission flow** — Create a helper `ensurePermissions()` inside useRecording.ts or as a separate utility:
```typescript
import {
  checkMicrophonePermission,
  requestMicrophonePermission,
  checkScreenRecordingPermission,
  requestScreenRecordingPermission,
} from "tauri-plugin-macos-permissions-api";
```

Flow:
1. Check mic permission. If not granted, request it. If still denied, show inline error and return false.
2. Check screen recording permission. If not granted:
   a. Show a pre-explanation dialog FIRST (locked decision): "We need Screen Recording access to capture meeting audio from apps like Zoom/Meet" — use a simple `window.confirm()` or a small React dialog. This is critical UX because "Screen Recording" sounds unrelated to audio.
   b. Call `requestScreenRecordingPermission()` — this does NOT show a native dialog on macOS. It silently adds the app to System Settings.
   c. Show inline guidance: "Please enable openNotes in System Settings > Privacy > Screen Recording, then restart the app." Include a button that calls `open("x-apple.systempreferences:com.apple.preference.security?Privacy_ScreenCapture")` via `@tauri-apps/plugin-shell`.
   d. Return false (app needs restart after granting Screen Recording).
3. If both granted, return true.

Note: Mic permission DOES show a native macOS dialog via `requestMicrophonePermission()`. Screen recording does NOT — it only adds to System Settings. This asymmetry must be handled correctly in the UI messaging.

**If permission denied** — The RecordView should show a permission status section with:
- Green checkmark if mic is granted, red X if not
- Green checkmark if screen recording is granted, amber warning if not (recording works mic-only)
- "Open System Settings" button if screen recording not granted
